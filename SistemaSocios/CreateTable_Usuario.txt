-- Tabela Usuario
CREATE TABLE IF NOT EXISTS Usuario (
    Id VARCHAR(36) PRIMARY KEY,
    nomeUsuario VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    senha VARCHAR(255),
    enderecoId VARCHAR(36),
    telefoneId VARCHAR(36),
    profissao VARCHAR(255),
    EscolaridadId VARCHAR(36),
    RedeSocialId VARCHAR(36),
    dataEntrada DATETIME,
    dataIniciacao DATETIME,
    dataUltimaObrigacao DATETIME,
    JurosMensalidadeId VARCHAR(36),
    ValorMensalidadeId VARCHAR(36),
    EntidadeId VARCHAR(36),
    perfilId VARCHAR(36),
    FOREIGN KEY (enderecoId) REFERENCES EnderecoUsuario(Id),
    FOREIGN KEY (telefoneId) REFERENCES TelefoneUsuario(Id),
    FOREIGN KEY (EscolaridadId) REFERENCES EscolaridadeUsuario(Id),
    FOREIGN KEY (RedeSocialId) REFERENCES RedeSocialUsuario(Id),
    FOREIGN KEY (JurosMensalidadeId) REFERENCES JurosMensalidade(Id),
    FOREIGN KEY (ValorMensalidadeId) REFERENCES ValorMensalidade(Id),
    FOREIGN KEY (EntidadeId) REFERENCES Entidade(Id),
    FOREIGN KEY (perfilId) REFERENCES perfil(Id)
);

-- Tabela EnderecoUsuario
CREATE TABLE IF NOT EXISTS EnderecoUsuario (
    Id VARCHAR(36) PRIMARY KEY,
    Endereco VARCHAR(255),
    Cidade VARCHAR(255),
    Estado VARCHAR(255),
    TipoEnderecoUsuarioId VARCHAR(36),
    Cep VARCHAR(255),
    FOREIGN KEY (TipoEnderecoUsuarioId) REFERENCES TipoEnderecoUsuario(Id)
);

-- Tabela TelefoneUsuario
CREATE TABLE IF NOT EXISTS TelefoneUsuario (
    Id VARCHAR(36) PRIMARY KEY,
    DddTelefoneUsuario VARCHAR(10),
    NumeroTelefoneUsuario VARCHAR(20),
    TipoTelefoneId VARCHAR(36),
    FOREIGN KEY (TipoTelefoneId) REFERENCES TipoTelefoneUsuario(Id)
);

-- Tabela TipoEnderecoUsuario
CREATE TABLE IF NOT EXISTS TipoEnderecoUsuario (
    Id VARCHAR(36) PRIMARY KEY,
    Descricao VARCHAR(255)
);

-- Tabela EscolaridadeUsuario
CREATE TABLE IF NOT EXISTS EscolaridadeUsuario (
    Id VARCHAR(36) PRIMARY KEY,
    DescricaoEscolaridade VARCHAR(255) NOT NULL,
    StatusEscolaridade BOOLEAN NOT NULL
);

-- Tabela RedeSocialUsuario
CREATE TABLE IF NOT EXISTS RedeSocialUsuario (
    Id VARCHAR(36) PRIMARY KEY,
    DescricaoRedeSocialUsuario VARCHAR(255),
    Descricao VARCHAR(255),
    UrlRedeSocial VARCHAR(255),
    IsActive BOOLEAN
);

-- Tabela JurosMensalidade
CREATE TABLE IF NOT EXISTS JurosMensalidade (
    Id VARCHAR(36) PRIMARY KEY,
    ValorJuros DECIMAL(18, 15),
    DataVigenciaJuros DATETIME,
    StatusValorJuros BOOLEAN
);

-- Tabela ValorMensalidade
CREATE TABLE IF NOT EXISTS ValorMensalidade (
    Id VARCHAR(36) PRIMARY KEY,
    ValorMensalidade DECIMAL(18, 15),
    VigenciaMensalidade DATETIME,
    StatusValorMensalidade BOOLEAN
);

-- Tabela Entidade
CREATE TABLE IF NOT EXISTS Entidade (
    Id VARCHAR(36) PRIMARY KEY,
    DescricaoEntidade VARCHAR(255),
    StatusEntidade BOOLEAN
);

-- Tabela perfil
CREATE TABLE IF NOT EXISTS perfil (
    Id VARCHAR(36) PRIMARY KEY,
    DescricaoPerfil VARCHAR(255),
    StatusPerfil BOOLEAN
);
